创建 `src/stores/theme.ts` 文件，使用 Pinia Store 模式管理主题状态：

**Store 结构：**
参考 `src/stores/game.ts` 的模式，使用 `defineStore` + setup 函数模式：

```typescript
import { ref, computed } from 'vue'
import { defineStore } from 'pinia'
import type { ThemeId, ThemeConfig } from '@/core/theme'
import { THEMES } from '@/core/theme'

export const useThemeStore = defineStore('theme', () => {
  // ===== 状态 =====
  /** 当前激活的主题 ID（默认为 'neon'） */
  const currentThemeId = ref<ThemeId>('neon')

  // ===== 计算属性 =====
  /** 当前主题配置对象 */
  const currentTheme = computed<ThemeConfig>(() => THEMES[currentThemeId.value])

  // ===== 操作方法 =====
  /**
   * 设置主题
   * @param themeId 主题 ID
   * @throws {Error} 如果主题 ID 不存在
   */
  function setTheme(themeId: ThemeId) {
    if (!THEMES[themeId]) {
      throw new Error(`Unknown theme: ${themeId}`)
    }
    currentThemeId.value = themeId
    // DOM 更新在 useTheme Composable 中处理
  }

  return {
    currentThemeId,
    currentTheme,
    setTheme
  }
})
```

**关键规则：**
- 使用 `ref&lt;ThemeId&gt;` 确保类型安全（字面量联合类型）
- 使用 `computed` 派生当前主题配置对象
- setTheme 方法验证主题 ID 存在性（运行时检查）
- **不直接操作 DOM**（DOM 操作在 Composable 中处理，保持 Store 纯粹）
- **不实现持久化**（持久化在 Phase 5 添加，避免混合关注点）
- 添加完整的 JSDoc 注释（中文）

**与现有 game.ts 的区别：**
- game.ts 在初始化时从 localStorage 加载状态
- theme.ts **不**加载持久化状态（Phase 5 实现）
- theme.ts **不**直接操作 DOM（Composable 负责）
