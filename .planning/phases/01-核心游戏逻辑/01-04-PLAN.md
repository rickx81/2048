export const useCounterStore = defineStore('counter', () => {

  const doubleCount = computed(() => count.value * 2)
  function increment() {
    count.value++
  }

  return { count, doubleCount, increment }
})
```

import { defineStore } from 'pinia'
import type { Grid, Direction, GameStatus } from '@/core/types'
import { createEmptyGrid, createInitialGrid } from '@/core/utils'
import { move, isGameOver, isGameWon } from '@/core/game'

export const useGameStore = defineStore('game', () => {
  // 状态
  const grid = ref<Grid>(createEmptyGrid())
  const score = ref<number>(0)
  const status = ref<GameStatus>(GameStatus.IDLE)
  const history = ref<Grid[]>([])

  // 计算属性
  const isGameOver = computed<boolean>(() => isGameOver(grid.value))
  const isGameWon = computed<boolean>(() => isGameWon(grid.value))

  // 操作方法
  function initialize() {
    grid.value = createInitialGrid()
    score.value = 0
    status.value = GameStatus.PLAYING
    history.value = []
  }

  function moveGrid(direction: Direction) {
    // 实现在下一个任务
  }

  function reset() {
    grid.value = createEmptyGrid()
    score.value = 0
    status.value = GameStatus.IDLE
    history.value = []
  }

  return {
    // 状态
    grid,
    score,
    status,
    history,

    // 计算属性
    isGameOver,
    isGameWon,

    // 方法
    initialize,
    moveGrid,
    reset
  }
})
```

```typescript
function moveGrid(direction: Direction) {
  // 如果游戏已结束，不允许移动
  if (status.value === GameStatus.LOST) {
    return
  }

  // 保存当前状态到历史记录（为 Phase 2 撤销功能预留）
  history.value.push([...grid.value])

  // 调用核心移动函数
  const result = move(grid.value, direction)

  // 如果没有发生移动，移除历史记录并返回
  if (!result.moved) {
    history.value.pop()
    return
  }

  // 更新网格和分数
  grid.value = result.grid
  score.value += result.score

  // 更新游戏状态
  if (isGameWon(grid.value) && status.value !== GameStatus.WON) {
    status.value = GameStatus.WON
  } else if (isGameOver(grid.value)) {
    status.value = GameStatus.LOST
  }
}
```

要点：
- 游戏结束后（LOST）不允许移动
- 移动前保存历史记录
- 无效移动时移除历史记录（保持一致性）
- 有效移动时更新状态
- 检测胜利和结束，更新 status
- 胜利后可继续移动（status 保持 WON）
